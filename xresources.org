** Updated Theme Switcher Integration

Add this to your `~/.local/bin/theme-switcher` in the `apply_xterm_theme` function:

```bash
# Function to apply XTerm theme
apply_xterm_theme() {
    local theme="$1"
    local theme_dir="$THEMES_DIR/$theme"
    
    if [ -f "$theme_dir/Xresources" ]; then
        # Copy theme colors to config directory
        mkdir -p ~/.config
        cp "$theme_dir/Xresources" ~/.config/Xresources
        
        # Reload Xresources if X is running
        if command -v xrdb >/dev/null && [ -n "$DISPLAY" ]; then
            xrdb -merge ~/.Xresources
            echo "  ✅ XTerm colors applied and reloaded"
        else
            echo "  ✅ XTerm colors applied"
        fi
    else
        echo "  ❌ XTerm theme file not found for $theme"
    fi
}
